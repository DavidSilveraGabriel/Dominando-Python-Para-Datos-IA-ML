Módulo 3: Soluciones de Ejercicios NumPy

1Creación y Atributos

import numpy as np

vector = np.arange(5, 16)
matriz = np.ones((3, 4), dtype=int)
print(f"Vector:\n{vector}")
print(f"Matriz:\n{matriz}")
print(f"Matriz ndim: {matriz.ndim}") # 2
print(f"Matriz shape: {matriz.shape}") # (3, 4)
print(f"Matriz size: {matriz.size}")   # 12
print(f"Matriz dtype: {matriz.dtype}") # int64 o int32

2Indexación y Slicing

print(f"Primer elemento vector: {vector[0]}") # 5
print(f"Último elemento vector: {vector[-1]}")# 15
sub_vector = vector[2:7]
print(f"Sub-vector [2:7]: {sub_vector}") # [7 8 9 10 11]
print(f"Elemento matriz[1, 2]: {matriz[1, 2]}") # 1
sub_matriz = matriz[0:2, -2:]
print(f"Sub-matriz (primeras 2 filas, últimas 2 cols):\n{sub_matriz}")
ultima_columna = matriz[:, -1]
print(f"Última columna:\n{ultima_columna}") # [1 1 1]

3Operaciones Vectorizadas

a = np.array([1, 2, 3])
b = np.array([10, 20, 30])
print(f"a + b: {a + b}") # [11 22 33]
print(f"a - b: {a - b}") # [ -9 -18 -27]
print(f"a * b: {a * b}") # [ 10  40  90]
print(f"a / b: {a / b}") # [0.1 0.1 0.1]
print(f"a al cuadrado: {a**2}") # [1 4 9]
print(f"Raíz cuadrada de b: {np.sqrt(b)}")
print(f"¿Todos a < b?: {(a < b).all()}") # True

4Broadcasting

m = np.arange(1, 10).reshape((3, 3))
v = np.array([100, 200, 300])
c = np.array([[10], [20], [30]])
print(f"m + v:\n{m + v}")
print(f"m * c:\n{m * c}")

5Métodos Estadísticos

print(f"Suma total de m: {m.sum()}") # 45
print(f"Media total de m: {m.mean()}") # 5.0
print(f"Máximo de m: {m.max()}") # 9
print(f"Suma por columnas (axis=0): {m.sum(axis=0)}") # [12 15 18]
print(f"Mínimo por filas (axis=1): {m.min(axis=1)}")   # [1 4 7]

6Indexación Booleana

m_orig = np.arange(1, 10).reshape((3, 3)) # Reiniciar m
condicion = m_orig > 5
print(f"Condición (m > 5):\n{condicion}")
print(f"Elementos > 5: {m_orig[condicion]}") # [6 7 8 9]
m_modif = m_orig.copy() # Copiar para no afectar m original
m_modif[m_modif % 2 == 0] = 0
print(f"Matriz m con pares a 0:\n{m_modif}")
